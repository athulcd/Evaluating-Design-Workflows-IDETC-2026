verification_plan:
  requirement_verification:
    - requirement_id: R1
      method: demonstration
      procedure_summary: |-
        Set up the tool with a brake shoe, lining, and rivet in the workholding fixture.
        Actuate the tool to form the rivet head.
        Inspect the formed rivet for closure dimension and joint integrity.
      conditions: Tool mounted on bench; brake shoe/lining/rivet per spec; operator applies hand force.
      metric: Successful rivet set without damage to components.
      acceptance_criteria: All rivets set to target closure (~4 mm) with no cracks or damage; joint holds under light pull test.
      linked_components: [B1, B2, B3, B5]
    - requirement_id: R2
      method: inspection
      procedure_summary: |-
        Measure the maximum brake shoe diameter the tool can accommodate.
        Verify the throat depth and fixture geometry allow D = 250-400 mm.
      conditions: Tool set up on bench; use gauge blocks or sample shoes.
      metric: Supported brake shoe sizes.
      acceptance_criteria: Tool accommodates D = 250-400 mm without interference.
      linked_components: [B1, B5]
    - requirement_id: R3
      method: inspection

      acceptance_criteria: Tool accommodates B = 80-400 mm.
      linked_components: [B1, B5]
    - requirement_id: R4
